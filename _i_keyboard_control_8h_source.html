27 IKeyboardControl is a (musical) keyboard for IPlug instruments. The keyboard
28 starts and ends at C. Only depressed keys are drawn by this control, so the
29 entire keyboard (with all its keys released) should already be visible (e.g.
30 because it's in the background).
31 
32 By default pRegularKeys should contain 6 bitmaps (C/F, D, E/B, G, A, high
33 C), while pSharpKey should only contain 1 bitmap (for all flat/sharp keys).
34 If you want to use more bitmaps, e.g. if the 1st octave is alternatively
35 colored, then you should override the DrawKey() method.
36 
37 pKeyCoords should contain the x-coordinates of each key relative to the
38 start of the octave. (Note that only the coordinates for the flat/sharp keys
39 are actually used, the coordinates for the "regular" keys are ignored.)
40 
41 Here is code snippet defining a 4-octave keyboard starting at MIDI note 48
42 (C3):
43 
44 IBitmap regular = graphics.LoadIBitmap(REGULAR_KEYS_ID, REGULAR_KEYS_PNG, 6);
45 IBitmap sharp = graphics.LoadIBitmap(SHARP_KEY_ID, SHARP_KEY_PNG);
46 
47 // C# D# F# G# A#
48 int coords[12] = { 0, 13, 23, 39, 46, 69, 82, 92, 107, 115, 131, 138 };
49 
50 // Store a pointer to the keyboard in member variable IControl* mKeyboard
51 mKeyboard = new IKeyboardControl(this, x, y, 48, 4, &regular, &sharp, coords);
52 
53 graphics.AttachControl(mKeyboard);

56 can pull status information from the plug-in, and send MIDI Note On/Off
57 message to the plug-in:
58 
59 // Should return non-zero if one or more keys are playing.
60 int MyPlug::GetNumKeys()
61 {
62  return mNumKeys;
63 }
64 
65 // Should return true if the specified key is playing.
66 bool MyPlug::GetKeyStatus(int key)
67 {
68  return mKeyStatus[key];

70 
71 (Instead of int you can also use any other integer types for the
72 GetNumKeys() and GetKeyStatus() methods, i.e. "char GetNumKeys()" or
73 "bool GetKeyStatus(char note)" will also work.)
74 
75 When the keyboard should be redrawn, e.g. when the plug-in has received a

148  // Draws only the keys that are currently playing.
149  virtual bool Draw(IGraphics& graphics)
150  {
151  // Skip if no keys are playing.
152  if (((PLUG_CLASS_NAME*)mPlug)->GetNumKeys() == 0 && mKey == -1) return true;
153 
154  // "Regular" keys
155  IRECT r(mRECT.L, mRECT.T, mRECT.L + mRegularKeys.W, mRECT.B);
156  int key = 0;
157  while (key < mMaxKey)
158  {
159  // Draw the key.
160  int note = key % 12;
161  if (((PLUG_CLASS_NAME*)mPlug)->GetKeyStatus(key + mMinNote) || key == mKey) DrawKey(pGraphics, &r, key, note, false);
162 
163  // Next, please!
164  key += mNextKey[note];
165  r.L += mRegularKeys.W;
166  r.R += mRegularKeys.W;
167  }
168  // Draw the high C.
169  if (((PLUG_CLASS_NAME*)mPlug)->GetKeyStatus(key + mMinNote) || key == mKey) DrawKey(pGraphics, &r, key, 0, false);
170 
171  // Flat/sharp keys
172  int l = mRECT.L;
173  r.B = mRECT.T + mSharpKey.H / mSharpKey.N;
174  key = 1;
175  while (true)
176  {
177  // Draw the key.
178  int note = key % 12;
179 
180  r.L = l + mKeyCoords[note];
181  r.R = r.L + mSharpKey.W;
182  if (((PLUG_CLASS_NAME*)mPlug)->GetKeyStatus(key + mMinNote) || key == mKey) DrawKey(pGraphics, &r, key, note, true);
183 
184  // Next, please!
185  key += mNextKey[note];
186  if (key > mMaxKey) break;
187  if (note == 10) l += mOctaveWidth;
188  }
189 
190  return true;
191  }

229  // Returns the key number at the (x, y) coordinates.
230  int GetMouseKey(int x, int y)
231  {
232  // Skip if the coordinates are outside the keyboard's rectangle.
233  if (x < mTargetRECT.L || x >= mTargetRECT.R || y < mTargetRECT.T || y >= mTargetRECT.B) return -1;
234  x -= mTargetRECT.L;
235  y -= mTargetRECT.T;
236 
237  // Calculate the octave.
238  int octave = x / mOctaveWidth;
239  x -= octave * mOctaveWidth;
240 
241  // Flat/sharp key
242  int note;
243  int h = mSharpKey.H / mSharpKey.N;
244 
245  if (y < h && octave < mNumOctaves)
246  {
247  // C#
248  if (x < mKeyCoords[1]) goto RegularKey;
249  if (x < mKeyCoords[1] + mSharpKey.W)
250  {
251  note = 1;
252  goto CalcVelocity;
253  }
254  // D#
255  if (x < mKeyCoords[3]) goto RegularKey;
256  if (x < mKeyCoords[3] + mSharpKey.W)
257  {
258  note = 3;
259  goto CalcVelocity;
260  }
261  // F#
262  if (x < mKeyCoords[6]) goto RegularKey;
263  if (x < mKeyCoords[6] + mSharpKey.W)
264  {
265  note = 6;
266  goto CalcVelocity;
267  }
268  // G#
269  if (x < mKeyCoords[8]) goto RegularKey;
270  if (x < mKeyCoords[8] + mSharpKey.W)
271  {
272  note = 8;
273  goto CalcVelocity;
274  }
275  // A#
276  if (x < mKeyCoords[10]) goto RegularKey;
277  if (x < mKeyCoords[10] + mSharpKey.W)
278  {
279  note = 10;
280  goto CalcVelocity;
281  }
282  }
283 
284  RegularKey:
285  {
286  h = mRECT.H();
287  int n = x / mRegularKeys.W;
288  note = n * 2;
289  if (n >= 3) note--;
290  }
291 
292  CalcVelocity:
293  // Calculate the velocity depeding on the vertical coordinate
294  // relative to the key height.
295  mVelocity = (double)y / (double)h;
296 
297  return note + octave * 12;
298  }
