278 bool IGraphicsLice::DrawIText(const IText& text, const char* str, IRECT& rect, bool measure)
279 {
280  if (!str || str[0] == '\0')
281  {
282  return true;
283  }
284 
285  LICE_IFont* font = text.mCached;
286 
287  if (!font || text.mCachedScale != mDisplayScale)
288  {
289  font = CacheFont(text, mDisplayScale);
290  if (!font) return false;
291  }
292 
293  LICE_pixel color = LiceColor(text.mColor);
294  font->SetTextColor(color);
295 
296  UINT fmt = DT_NOCLIP;
297  if (LICE_GETA(color) < 255) fmt |= LICE_DT_USEFGALPHA;
298  if (text.mAlign == IText::kAlignNear)
299  fmt |= DT_LEFT;
300  else if (text.mAlign == IText::kAlignCenter)
301  fmt |= DT_CENTER;
302  else // if (text.mAlign == IText::kAlignFar)
303  fmt |= DT_RIGHT;
304 
305  if (measure)
306  {
307  fmt |= DT_CALCRECT;
308  RECT R = {0,0,0,0};
309  font->DrawText(mDrawBitmap, str, -1, &R, fmt);
310 
311  if( text.mAlign == IText::kAlignNear)
312  {
313  rect.R = R.right;
314  }
315  else if (text.mAlign == IText::kAlignCenter)
316  {
317  rect.L = (int) rect.MW() - (R.right/2);
318  rect.R = rect.L + R.right;
319  }
320  else // (text.mAlign == IText::kAlignFar)
321  {
322  rect.L = rect.R - R.right;
323  rect.R = rect.L + R.right;
324  }
325 
326  rect.B = rect.T + R.bottom;
327  }
328  else
329  {
330  IRECT r = rect;
331  r.Scale(mDisplayScale);
332  RECT R = { r.L, r.T, r.R, r.B };
333  font->DrawText(mDrawBitmap, str, -1, &R, fmt);
334  }
335 
336  return true;

339 LICE_IFont* IGraphicsLice::CacheFont(const IText& text, double scale)
340 {
341  WDL_String hashStr(text.mFont);
342  hashStr.AppendFormatted(50, "-%d-%d-%d", text.mSize, text.mOrientation, text.mStyle);
343 
344  LICE_CachedFont* font = (LICE_CachedFont*)s_fontCache.Find(hashStr.Get(), scale);
345  if (!font)
346  {
347  font = new LICE_CachedFont;
348  int h = round(text.mSize * scale);
349  int esc = 10 * text.mOrientation;
350  int wt = (text.mStyle == IText::kStyleBold ? FW_BOLD : FW_NORMAL);
351  int it = (text.mStyle == IText::kStyleItalic ? TRUE : FALSE);
352 
353  int q;
354  if (text.mQuality == IText::kQualityDefault)
355  q = DEFAULT_QUALITY;
356 #ifdef CLEARTYPE_QUALITY
357  else if (text.mQuality == IText::kQualityClearType)
358  q = CLEARTYPE_QUALITY;
359  else if (text.mQuality == IText::kQualityAntiAliased)
360 #else
361  else if (text.mQuality != IText::kQualityNonAntiAliased)
362 #endif
363  q = ANTIALIASED_QUALITY;
364  else // if (text.mQuality == IText::kQualityNonAntiAliased)
365  q = NONANTIALIASED_QUALITY;
366 
367 #ifdef OS_OSX
368  bool resized = false;
369  Resize:
370  if (h < 2) h = 2;
371 #endif
372  HFONT hFont = CreateFont(h, 0, esc, esc, wt, it, FALSE, FALSE, ANSI_CHARSET, OUT_DEFAULT_PRECIS, CLIP_DEFAULT_PRECIS, q, DEFAULT_PITCH, text.mFont);
373  if (!hFont)
374  {
375  delete(font);
376  return 0;
377  }
378  font->SetFromHFont(hFont, LICE_FONT_FLAG_OWNS_HFONT | LICE_FONT_FLAG_FORCE_NATIVE);
379 #ifdef OS_OSX
380  if (!resized && font->GetLineHeight() != h)
381  {
382  h = int((double)(h * h) / (double)font->GetLineHeight() + 0.5);
383  resized = true;
384  goto Resize;
385  }
386 #endif
387  s_fontCache.Add(font, hashStr.Get(), scale);
388  }
389  text.mCached = font;
390  text.mCachedScale = scale;
391  return font;
392 }
